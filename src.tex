The activation of the $n^{\text{th}}$ neuron (or, more accurately in this case, layer) is the product of the ${(n-1)}^{\text{th}}$ weight matrix $\mathbf{W}$ and prior activation $\vec{a}.$ Let $\vec{a}_n$ be the output layer, and let $\vec{a}_1$ be the input layer.

Let:
% n-1 mapping
\begin{align*}
\underbrace{\mathbf{W}_{n-1}}_{q \times p}: \underbrace{\vec{a}_{n-1}}_{p \times 1} \mapsto \underbrace{\vec{a}_{n}}_{q \times 1}.
\end{align*}

Let the cost function be defined as:

% cost function def
\begin{align*}
C&=(\vec{a}_n-\vec{e})^2 \\
&=(\sigma(\mathbf{W}_{n-1}\vec{a}_{n-1}) - \vec{e})^2 \\
&=(\sigma(\mathbf{W}_{n-1} \sigma(\mathbf{W}_{n-2}\vec{a}_{n-2}) ) - \vec{e})^2 \\
&\vdotswithin{=}
\end{align*}
Thus:
% first derivative
\begin{align*}
\underbrace{\frac{\partial C}{\partial \mathbf{W}_{n-1}}}_{q \times p}
&=
\underbrace{2(\sigma(\mathbf{W}_{n-1}\vec{a}_{n-1})-\vec{e})}_{q \times 1} \odot 
\underbrace{\sigma'(\mathbf{W}_{n-1}\vec{a}_{n-1})}_{q \times 1}
\underbrace{\vec{a}_{n-1}^{\text{T}}}_{1 \times p} \\
&=\delta_{n-1}\vec{a}_{n-1}^{\text{T}}.
\end{align*}

% consider n-2
Consider $n-2$:
\begin{align*}
\underbrace{\mathbf{W}_{n-2}}_{p \times r}: \underbrace{\vec{a}_{n-2}}_{r \times 1} \mapsto \underbrace{\vec{a}_{n-1}}_{p \times 1}.
\end{align*}

% work n-2
\begin{align*}
\frac{\partial C}{\partial \mathbf{W}_{n-2}}
&=
2(\sigma(\mathbf{W}_{n-1}\vec{a}_{n-1})-\vec{e}) \odot 
\sigma'(\mathbf{W}_{n-1}\vec{a}_{n-1})
\frac{\partial \left[ \mathbf{W}_{n-1}\vec{a}_{n-1} \right]}{\partial \mathbf{W}_{n-2}} \\
&=
2(\sigma(\mathbf{W}_{n-1}\vec{a}_{n-1})-\vec{e}) \odot 
\sigma'(\mathbf{W}_{n-1}\vec{a}_{n-1})
\frac{\partial[\mathbf{W}_{n-1}\sigma(\mathbf{W}_{n-2}\vec{a}_{n-2})]}{\partial\mathbf{W}_{n-2}} \\
&=
2(\sigma(\mathbf{W}_{n-1}\vec{a}_{n-1})-\vec{e}) \odot 
\sigma'(\mathbf{W}_{n-1}\vec{a}_{n-1})
\mathbf{W}_{n-1}
\sigma'(\mathbf{W}_{n-2}\vec{a}_{n-2})
\vec{a}_{n-2}
\end{align*}
This is quite messy, but the chain rule is done properly. Then, correct for dimensions:

% n-2 dimensions
\begin{align*}
\frac{\partial C}{\partial \mathbf{W}_{n-2}} &=
\underbrace{2(\sigma(\mathbf{W}_{n-1}\vec{a}_{n-1})-\vec{e})}_{q \times 1}
\odot 
\underbrace{\sigma'(\mathbf{W}_{n-1}\vec{a}_{n-1})}_{q \times 1}
\underbrace{\mathbf{W}_{n-1}}_{q \times p}
\underbrace{\sigma'(\mathbf{W}_{n-2}\vec{a}_{n-2})}_{p \times 1}
\underbrace{\vec{a}_{n-2}}_{r \times 1} \\
&=
\left[ \underbrace{\mathbf{W}_{n-1}}_{p \times q}^{\text{T}}
\underbrace{2(\sigma(\mathbf{W}_{n-1}\vec{a}_{n-1})-\vec{e})}_{q \times 1}
\odot 
\underbrace{\sigma'(\mathbf{W}_{n-1}\vec{a}_{n-1})}_{q \times 1} \right]
\odot
\underbrace{\sigma'(\mathbf{W}_{n-2}\vec{a}_{n-2})}_{p \times 1}
\underbrace{\vec{a}_{n-2}}_{1 \times r}^{\text{T}} \\
\Rightarrow
\underbrace{\frac{\partial C}{\partial \mathbf{W}_{n-2}}}_{p \times r} &=
\mathbf{W}_{n-1}^{\text{T}}\delta_{n-1}\odot \sigma'(\mathbf{W}_{n-2}\vec{a}_{n-2})
\vec{a}_{n-2}^{\text{T}} \\
&= \delta_{n-2}\vec{a}_{n-2}^{\text{T}}
\end{align*}

Or, in general:
$$\frac{\partial C}{\partial \mathbf{W}_{j}} = \delta_{j}\vec{a}_{j}^{\text{T}},$$
where,
$$\delta_{i}=\begin{cases} 2(\sigma(\mathbf{W}_i\vec{a}_i) - \vec{e})\odot\sigma'(\mathbf{W}_i \vec{a}_i) & i=n-1 \\
\mathbf{W}_{i+1}^{\text{T}}\delta_{i+1}\odot\sigma'(\mathbf{W}_i\vec{a}_i) & i<n-1.
\end{cases}$$
